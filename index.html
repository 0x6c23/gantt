Gantt

A simple, interactive, modern gantt chart library for the web

View on GitHub
Simple Gantt Chart
Usage
	          
  var gantt = new Gantt({
    parent_selector: '#gantt',
    tasks: [
      {
        start: "2016-10-04",
        end: "2016-10-10",
        name: "Explore ERPNext",
        id: 0,
        progress: 30
      },
      ...
    ],
    date_format: "YYYY-MM-DD"
  });
  gantt.render();
Change view modes
Usage
	          
  var gantt = new Gantt({
    ...
    ...
  });
  gantt.render();
  gantt.set_viewmode('Half Day');
Set events
Usage
	          
  var gantt = new Gantt({
    ...
    events: {
        bar_on_click: function (task) {
            console.log(task);
        },
        bar_on_date_change: function (task, start, end) {
            console.log(task, start, end);
        },
        bar_on_progress_change: function (task, progress) {
            console.log(task, progress);
        },
        on_viewmode_change: function (mode) {
            console.log(mode);
        }
    }
    ...
  });
  gantt.render();
